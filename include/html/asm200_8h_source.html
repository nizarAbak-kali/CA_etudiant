66  /* Mips instruction set: */
67  /* */
68  /* Opcods in lower case are Mips instructions */
69  /* Opcods in lower case and * are Mips 1 instructions */
70  /* Opcods in lower case and % are Mips 32 instructions */
71  /* Opcods in upper case are application specific */
72  /* */
73  /* primary opcod (31 downto 26) : */
74  /* | 0 1 2 3 4 5 6 7 */
75  /* --+-----+-----+-----+-----+-----+-----+-----+-----+ */
76  /* 0 |speci|bcond| j | jal | beq | bne |blez |bgtz | */
77  /* 1 |addi |addiu|slti |sltiu|andi | ori |xori | lui | */
78  /* 2 |cop0 | + |cop2 |spec2| + | + | + | + | */
79  /* 3 | | | | | + | + | | + | */
80  /* 4 | lb | lh | lwl | lw | lbu | lhu | lwr | | */
81  /* 5 | sb | sh | + | sw | | | + | + | */
82  /* 6 | ll | + | + |pref | | + | + | | */
83  /* 7 | sc | + | + | + | | + | + | | */
84  /* */
85  /* */
86  /* special opcod extension (5 downto 0) : */
87  /* | 0 1 2 3 4 5 6 7 */
88  /* --+-----+-----+-----+-----+-----+-----+-----+-----+ */
89  /* 0 | *** | + | srl | sra |sllv | |srlv |srav | */
90  /* 1 | jr |jalr | + | + |sysca|break| |sync | */
91  /* 2 |mfhi |mthi |mflo |mtlo | | | | | */
92  /* 3 |mult |multu| div |divu | | | | | */
93  /* 4 | add |addu | sub |subu | and | or | xor | nor | */
94  /* 5 | | | slt |sltu | | | | | */
95  /* 6 | + | + | + | + | + | | + | | */
96  /* 7 | | | | | + | + | + | + | */
97  /* */

99  /* between sll */
